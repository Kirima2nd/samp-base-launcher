		private void OnPlayButtonClick(object sender, EventArgs e)
		{
			string ip = "samp.launcher-ip.net:7777 "; // ini ip
			string pw = "sadasdasd"; // ini password
			
			string sa_path = Registry.CurrentUser.OpenSubKey("Software\\\\SAMP").GetValue("gta_sa_exe").ToString();
			sa_path = sa_path.Substring(0, sa_path.LastIndexOf("\\") + 1);
			
			Registry.CurrentUser.OpenSubKey("Software\\SAMP", writable: true).SetValue("PlayerName", BoxName.Text);
			Process.Start(sa_path + "samp.exe", ip + pw);
		}
		
		private void OnFormLoad(object sender, EventArgs e)
		{
			string sa_path = Registry.CurrentUser.OpenSubKey("Software\\\\SAMP").GetValue("gta_sa_exe").ToString();
			sa_path = sa_path.Substring(0, sa_path.LastIndexOf("\\") + 1);
			
			BoxName.Text = Registry.CurrentUser.OpenSubKey("Software\\SAMP", writable: true).GetValue("PlayerName").ToString();
			
			if (!File.Exists(sa_path + "gta_sa.exe"))
			{
				MessageBox.Show("Tidak ada gta_sa.exe dan Folder GTA\n" + sa_path + "\nReinstall client SAMP, atau reinstall GTA SA.", "Launcher | Error", MessageBoxButtons.OK, MessageBoxIcon.Hand);
				Close();
				Application.Exit();
			}
			else if (!File.Exists(sa_path + "samp.exe"))
			{
				MessageBox.Show("Buka samp.exe:\n" + sa_path + "\nInstall ulang SAMP.", "Launcher | Error", MessageBoxButtons.OK, MessageBoxIcon.Hand);
				Close();
				Application.Exit();
			}
		}
